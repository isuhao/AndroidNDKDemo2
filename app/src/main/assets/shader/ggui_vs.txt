//-- ggui_vs ---------------------------------------------------------------------------------------
precision highp float;
//--------------------------------------------------------------------------------------------------
//Vertex Struct
attribute vec3  myVertex;
attribute vec3  myUV;
attribute vec4  myColor;
//--------------------------------------------------------------------------------------------------
//Global Param
uniform mat4  g_matProjectView;
//--------------------------------------------------------------------------------------------------
//Output Struct
varying vec2  fragUV;
varying vec4  fragColor;
//--------------------------------------------------------------------------------------------------
void main()
{
    vec4 worldVertex = vec4(myVertex, 1);
    gl_Position = g_matProjectView * worldVertex;

    fragUV = myUV.xy;
    fragColor = myColor;
}
//--------------------------------------------------------------------------------------------------
