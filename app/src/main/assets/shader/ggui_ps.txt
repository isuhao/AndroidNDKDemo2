//-- ggui_ps ---------------------------------------------------------------------------------------
precision highp float;
//--------------------------------------------------------------------------------------------------
//Global Param
uniform sampler2D  g_Texture;
//--------------------------------------------------------------------------------------------------
//Input Struct
varying vec2  fragUV;
varying vec4  fragColor;
//--------------------------------------------------------------------------------------------------
void main()
{
	vec4 theColor = texture2D(g_Texture, fragUV);

    theColor.rgb = fragColor.rgb * fragColor.a + theColor.rgb * (1.0 - fragColor.a);

	gl_FragColor = theColor;
}
//--------------------------------------------------------------------------------------------------
